Vuex例子

src/store.js

import Vue from 'vue'
import Vuex from 'vuex'

Vue.use(Vuex)

export default new Vuex.Store({
  state: {
    count: 1
  },
  mutations: {
    add(state, n) {
      state.count = state.count + (n || 1); // 3. mutations中: 修改state
    }
  },
  actions: {
    action1(context, n) {
      context.commit('add', n) // 2. action中: context.commit, 提交修改
    }
  },
  getters: {
    anotherCount(state) {
      return state.count + 1
    }
  }
})

src/views/View1.vue

<template>
  <div>
    <div @click="test">view1: 点我 {{ this.$store.state.count }}</div>
  </div>
</template>

<script>
export default {
  name: 'View1',
  methods: {
    test () {
      this.$store.dispatch('action1', 5) // 1. vue中: store.dispatch, 发送action
    }
  }
}
</script>

src/views/View2.vue

<template>
  <div>
    <div @click="test">view2: 点我 {{ count }} {{ anotherCount }}</div>
  </div>
</template>

<script>
import { mapState, mapActions, mapGetters } from 'vuex'

export default {
  name: 'View2',
  computed: {
    ...mapState({
      count: state => state.count // 4. vue中: state映射到组件computed属性
    }),
    ...mapGetters([
      'anotherCount'
    ])
  },
  methods: {
    ...mapActions([
      'action1'
    ]),
    test () {
      this.action1(2)
    }
  }
}
</script>
