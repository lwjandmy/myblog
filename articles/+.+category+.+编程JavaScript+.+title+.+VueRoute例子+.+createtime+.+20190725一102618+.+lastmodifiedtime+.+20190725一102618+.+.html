new Vue({
  router,

}).$mount('#app')

router.js

import Vue from 'vue'
import Router from 'vue-router'

Vue.use(Router)

export default new Router({
  mode: 'hash',
  routes: [{
    path: '/view1',
    name: 'view1',
    props: true,
    component: () =>
      import( /* webpackChunkName: "view1" */ './views/View1.vue')
  }, {
    path: '/view2',
    name: 'view2',
    props: true,
    component: () =>
      import( /* webpackChunkName: "view2" */ './views/View2.vue')
  }]
})

App.vue

<template>
  <div>
    <div @click="onView1Click" style="cursor: pointer">查看页面1</div>
    <div @click="onView2Click" style="cursor: pointer">查看页面2</div>
    <div>Router页面:</div>
    <router-view/>
  </div>
</template>

<script>
export default {
  name: 'App',
  methods: {
    onView1Click: function () {
      this.$router.push({
        name: 'view1',
        params: {
          data: "data1"
        }
      })
    },
    onView2Click: function () {
      this.$router.push({
        name: 'view2',
        params: {
          data: "data2"
        }
      })
    }
  }
}
</script>
