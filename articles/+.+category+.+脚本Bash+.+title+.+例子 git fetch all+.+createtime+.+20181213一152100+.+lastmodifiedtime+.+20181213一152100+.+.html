fetch-all2.sh

function fetch() {
    pushd $1
    git fetch
    popd
}

find . -maxdepth 1 -type d | while read path; do
    if [ "." == "$path" ]; then
        continue
    fi
    if [ "./.idea" == "$path" ]; then
        continue
    fi
    fetch $path
done
