d3.hierarchy

对分层数据进行分层初始化

var data_hierarchy = d3.hierarchy(data, children_fn)

d3.hierarchy({
    "name": "name1",
    "child": [
        {
            "name": "name2"
        }
    ]
}, function (d) { return d.child });
返回:
{
    "data": {
        "name": "name1"
        "child": [
            {
                "name": "name2"
            }
        ]
    }
    "children": [
        {
            "data": ...
            "depth": 1
            "height": 0
            "parent": 
        }
    ]
    "depth": 0
    "height": 1
    "parent": null
}
返回的数据格式:
{
    data 当前节点的数据
    depth 0是根, 每向下一级加1
    height 0是叶, 每向上一级加1
    parent 根为null, 其它为父节点
    children 叶为undefined, 其它为子节点数组
}

d3.tree

创建默认配置的树形布局, 返回一个布局函数

var tree_fn = d3.tree()

将分层数据root进行树形布局

tree_fn(root)

d3.tree()(d3.hierarchy({
    "name": "name1",
    "child": [
        {
            "name": "name2"
        }
    ]
}, function (d) { return d.child }))
返回:
{
    原成员变量...
    "x": 节点x坐标

d3.tree 树形分层 helloworld

index.html

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>测试分层布局</title>
    <script src="js/d3.v4.js"></script>
    <script src="js/jquery.1.9.1.js"></script>
</head>
<body>

    <svg width="100%" height="700px">
    </svg>

    <script>
        function draw(data) {
            // 数据分层初始化, 得到根
            var root = d3.hierarchy(data, function (d) { return d.child });

            // 用默认配置, 得到树形分层函数
            var tree_fn = d3.tree();

            // 进行树形分层
            // x 根据同一depth的节点数, 均分区间[0,1]. 如第一个节点, x=0; 第二个节点, x=0.5; 第三个节点, x=1
            // y 根据树的depth, 均分区间[0,1]. 如depth=0, y=0; depth=1, y=0.5; depth=2, y=1
            root = tree_fn(root);

            // 绘图
            d3.select('svg')
                .selectAll('circle')
                .data(root.descendants())
                .enter()
                .append('circle')
                .attr('i', function (d, i) { return i })
                .attr('cx', function (d, i) { return d.x * 100 })
                .attr('cy', function (d, i) { return d.y * 100 })
                .attr('r', function (d) { return 3 });
        }

        d3.json('data.json', draw);
    </script>
</body>
</html>
