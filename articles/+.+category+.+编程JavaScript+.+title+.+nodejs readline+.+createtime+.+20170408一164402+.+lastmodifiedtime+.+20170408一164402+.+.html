nodejs readline

readline用于nodejs中从终端读取用户输入的数据

以下例子展示readline模块用法

const readline = require('readline');
const rl = readline.createInterface(process.stdin, process.stdout);

rl.setPrompt('输入pdm文件路径, 输入空退出程序> ');
rl.prompt();

rl.on('line', function (line) {

    if (0 == line.length) {
        process.exit(0);
    }
    processPdm(line);

}).on('close', function () {
    process.exit(0);
});
