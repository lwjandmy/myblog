readline读取一行_promise格式

async function readLine() {
    return new Promise(resolve => {
        const dataList = []
        process.stdin.on('readable', () => {
            let chunk
            while ((chunk = process.stdin.read()) !== null) {
                dataList.push(chunk)
            }
        })
        process.stdin.on('end', () => {
            resolve(dataList.join(''))
        })
    })
}
