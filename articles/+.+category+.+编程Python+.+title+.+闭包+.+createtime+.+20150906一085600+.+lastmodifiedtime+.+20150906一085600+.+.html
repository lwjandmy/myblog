闭包

def bag(n):
    def get():
        nonlocal n
        n = n - 1
        return n
    return get      # get记住了外部函数(bag)的参数n (n的引用计数为2, bag函数结束后, n引用计数为1, 未回收)

abag=bag(2)         # 工厂函数生成一个函数给f
abag()              # output: 1
abag()              # output: 0
