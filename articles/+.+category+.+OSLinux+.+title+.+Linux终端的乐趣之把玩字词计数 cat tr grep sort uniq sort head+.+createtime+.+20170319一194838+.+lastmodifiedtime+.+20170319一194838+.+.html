Linux终端的乐趣之把玩字词计数

计算一个文本文件中字和字符的出现频率

# 准备操作的文件
cd ~
LANG=en_US.UTF-8 man man > man.txt

# 统计出现频率最高的词
cat man.txt | tr ' '  '\012' | tr '[:upper:]' '[:lower:]' | tr -d '[:punct:]' | grep '[ a-z]' | sort | uniq -c | sort -r -n | head

# 统计出现频率最高的字母
cat man.txt | fold -w1 | grep '[ a-z]' | sort | uniq -c | sort -r -n
