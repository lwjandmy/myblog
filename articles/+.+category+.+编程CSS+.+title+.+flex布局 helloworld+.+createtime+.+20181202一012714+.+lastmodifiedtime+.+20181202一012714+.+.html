            /* 使用flex布局 */
            .flex {
                display: flex;

                min-height: 300px;
            }
            /* flex容器主轴方向: row row-reverse column column-reverse */
            .flex-direction {
                flex-direction: row-reverse;
            }
            /* flex容器内项目的换行方式: nowrap wrap wrap-reverse */
            .flex-wrap {
                flex-wrap: wrap;
            }
            /* flex容器内项目在主轴上的的对齐方式: flex-start flex-end center space-between space-around */
            .justify-content {
                justify-content: space-between;
            }
            /* flex容器内项目在交叉轴上的对齐方式: flex-start flex-end center baseline stretch */
            .align-items {
                align-items: stretch;
            }
            /* flex容器内项目在多跟交叉轴上的对齐方式: flex-start flex-end center space-between space-around stretch */
            .align-content {
                align-content: space-around;
            }
            /* flex项目排列次序, 决定排列顺序, 默认0 */
            .order {
                order: 1;
            }
            /* flex项目剩余主空间占用比例, 默认0 */
            .flex-grow {
                flex-grow: 1;
            }
            /* flex项目主空间缩小比例, 默认1 */
            .flex-shrink {
                flex-shrink: 0;
            }
            /* flex项目占用主轴空间, 默认auto, 根据内容自动计算 */
            .flex-basis {
                flex-basis: 100px;
            }
            /* flex项目在交叉轴上的对齐方式, 用于覆盖父容器的align-items, auto flex-start flex-end center baseline stretch */
            .align-self {
                align-self: flex-end;
            }
