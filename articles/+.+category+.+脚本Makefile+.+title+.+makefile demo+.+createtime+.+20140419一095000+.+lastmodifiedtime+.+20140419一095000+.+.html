makefile demo

# first example

COMPILE = g++ -c $< -o $@
LINK = g++ $^ -o $@

prog : 1.o 2.o 3.o
    $(LINK)

1.o : 1.cpp 1.h
    $(COMPILE)

2.o : 2.cpp 2.h
    $(COMPILE)

3.o : 3.cpp 3.h
    $(COMPILE)

$@ $^ $<含义:

  • $@ The file name of the target of the rule.
  • $^ The names of all the prerequisites, with spaces between them.
  • $< $< The name of the first prerequisite.

Example:

main : main1.o main2.o
$@ main
$^ main1.o main2.o
$< main1.o
