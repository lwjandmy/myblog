go 匿名类型 demo

func argsFunc(i int, args ...int) {
    fmt.Println(i)
    for i, v := range args {
        fmt.Println(i, v)
    }
}

func argsFunc2(args ...interface{}) {
    for _, arg := range args {
        switch arg.(type) {
        case int:
            fmt.Println("int: ", arg)
        case string:
            fmt.Println("string: ", arg)
        case int64:
            fmt.Println("int64: ", arg)
        default:
            fmt.Println("unknown type: ", arg)
        }
    }
}

func main() {
    argsFunc(1, 2, 3, 4, 5)
    argsFunc(1, []int{2, 3, 4, 5}...)
    argsFunc2(1, "2", int64(3), 4.5)
}
