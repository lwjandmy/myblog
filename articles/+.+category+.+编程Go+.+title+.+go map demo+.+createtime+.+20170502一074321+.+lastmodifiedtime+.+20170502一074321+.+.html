go map demo

func main() {
    // 创建map
    map1 := make(map[string] string)
    map2 := make(map[string] string, 100) // 设置map的初始容量为100

    // map元素赋值
    map1["key1"] = "value1"

    // 删除map元素
    delete(map1, "key1")

    // map元素查找
    v , ok := map1["key1"]
    if ok {
        fmt.Println("找到了", v)
    }

    fmt.Println(map1)
    fmt.Println(map2)
}
