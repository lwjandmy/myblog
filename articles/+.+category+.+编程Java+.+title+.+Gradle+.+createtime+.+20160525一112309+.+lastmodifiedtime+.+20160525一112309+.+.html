Gradle

初始化新工程

gradle init --type java-library

添加eclipse支持

配置build.gradle

apply plugin: 'eclipse'

gradle eclipse

Eclipse中导入工程

Eclipse: File -> Import..., General -> Existing Projects into Workspace.

设置Main-Class

jar {
    manifest {
        attributes 'Main-Class': 'org.test.HelloWorld'
    }
}

构建工程

gradle build

下载依赖

gradle dependencies

清理

gradle clean
