# settings.xml配置模板

```xml

<servers>
  <!-- 配置部署构件需要的账号密码 -->
  <server>
    <id>nexus-releases</id>
    <username>deployment</username>
    <password>deployment123</password>
  </server>
  <server>
    <id>nexus-snapshots</id>
    <username>deployment</username>
    <password>deployment123</password>
  </server>
</servers>

<profiles>
  <!-- 定义jdk版本 -->
  <profile>
    <id>jdk-1.6</id>
    <properties>
      <maven.compiler.source>1.6</maven.compiler.source>
      <maven.compiler.target>1.6</maven.compiler.target>
      <maven.compiler.compilerVersion>1.6</maven.compiler.compilerVersion>
    </properties>
  </profile>
  <profile>
    <id>jdk-1.7</id>
    <properties>
      <maven.compiler.source>1.7</maven.compiler.source>
      <maven.compiler.target>1.7</maven.compiler.target>
      <maven.compiler.compilerVersion>1.7</maven.compiler.compilerVersion>
    </properties>
  </profile>
  <profile>
    <id>jdk-1.8</id>
    <properties>
      <maven.compiler.source>1.8</maven.compiler.source>
      <maven.compiler.target>1.8</maven.compiler.target>
      <maven.compiler.compilerVersion>1.8</maven.compiler.compilerVersion>
    </properties>
  </profile>

  <!-- 新增nexus仓库指向私服, 修改中央仓库指向阿里云nexus -->
  <profile>
    <id>nexus</id>
    <repositories>
      <repository>
        <!-- 覆盖中央仓库配置 -->
        <id>central</id>
        <!-- 修改中央仓库地址为nexus仓库 -->
        <url>http://192.168.1.1:8081/nexus/content/groups/public/</url>
        <!-- 开启中央仓库的发布版和快照版的下载支持 -->
        <releases>
          <enabled>true</enabled>
        </releases>
        <snapshots>
          <enabled>true</enabled>
        </snapshots>
        <layout>default</layout>
      </repository>
      <!-- 增加阿里云镜像仓库配置 -->
      <repository>
        <id>Aliyun</id>
        <name>Aliyun Repository</name>
        <url>http://maven.aliyun.com/nexus/content/groups/public/</url>
      </repository>
      <!-- 原始maven中央仓库 -->
      <repository>
        <id>origin</id>
        <name>origin</name>
        <url>https://repo.maven.apache.org/maven2/</url>
      </repository>
      <!-- 原始maven中央仓库 -->
      <repository>
        <id>origin2</id>
        <name>origin2</name>
        <url>http://central.maven.org/maven2/</url>
      </repository>
      <!-- redhat仓库 -->
      <repository>
        <id>redhat-maven-repo</id>
        <name>redhat-maven-repo</name>
        <url>http://maven.repository.redhat.com/techpreview/all/</url>
      </repository>
    </repositories>
    <pluginRepositories>
      <pluginRepository>
        <!-- 覆盖中央仓库配置 -->
        <id>central</id>
        <!-- 修改中央仓库地址为nexus仓库 -->
        <url>http://192.168.1.1:8081/nexus/content/groups/public/</url>
        <!-- 开启中央仓库的发布版和快照版的下载支持 -->
        <releases>
          <enabled>true</enabled>
        </releases>
        <snapshots>
          <enabled>true</enabled>
        </snapshots>
        <layout>default</layout>
      </pluginRepository>
      <!-- 增加阿里云镜像仓库配置 -->
      <pluginRepository>
        <id>Aliyun</id>
        <name>Aliyun pluginRepository</name>
        <url>http://maven.aliyun.com/nexus/content/groups/public/</url>
      </pluginRepository>
      <!-- 原始maven中央仓库 -->
      <pluginRepository>
        <id>origin</id>
        <name>origin</name>
        <url>https://repo.maven.apache.org/maven2/</url>
      </pluginRepository>
      <!-- 原始maven中央仓库 -->
      <pluginRepository>
        <id>origin2</id>
        <name>origin2</name>
        <url>http://central.maven.org/maven2/</url>
      </pluginRepository>
      <!-- redhat仓库 -->
      <pluginRepository>
        <id>redhat-maven-repo</id>
        <name>redhat-maven-repo</name>
        <url>http://maven.pluginRepository.redhat.com/techpreview/all/</url>
      </pluginRepository>
    </pluginRepositories>
  </profile>

</profiles>

<!-- 使配置生效 -->
<activeProfiles>
  <activeProfile>jdk-1.6</activeProfile>
  <activeProfile>nexus</activeProfile>
</activeProfiles>

```
