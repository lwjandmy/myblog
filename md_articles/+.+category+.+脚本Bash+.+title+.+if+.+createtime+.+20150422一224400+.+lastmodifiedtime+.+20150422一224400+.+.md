# if

```bash
# -e   test file exist
if [[ -e "file_exist" ]]; then
    echo "file exist"
else
    echo "file not exist"
fi



# normal compare: < > <= >= ! ...
n1=1
n2=2
if [[ $n1 < $n2  ]]; then
    echo '1 < 2'
fi


# test string length == 0
emptyStr=""
if [[ -z $emptyStr ]]; then
    echo "empty string"
    echo -e "string length: #emptyStr"
fi


# get string length  看不懂的语法。。。
echo "string length: ${#emptystr}"
```
