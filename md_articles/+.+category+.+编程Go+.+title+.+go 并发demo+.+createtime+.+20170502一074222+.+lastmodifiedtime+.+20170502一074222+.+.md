# go 并发demo




main.go

```go
package main

import "fmt"

func Add(x int, y int, result chan int) {
	result <- x + y;
}

func main() {
	chs := make([]chan int, 10)
	for i := 0; i != 10; i++ {
		chs[i] = make(chan int)
		go Add(i, i, chs[i])
	}
	for _, ch := range chs {
		fmt.Println(<- ch)
	}
}
```
