# 文件操作

```Python
# 写入
f=open('data.txt','w')
f.write('Hello\n')
f.write('world\n')
f.close()

# 读取
f=open('data.txt')
text=f.read()
text.split()  output: ['Hello','world']

# open函数的参数
'r' read, 读取模式
'w' write,写入模式
'a' append,尾部追加
'b' binary, 二进制模式
'+' 读写模式，用于对文件的查找和修改操作

# 用文件迭代器一行一行扫描文件
for line in open('myfile'):....   open函数返回的对象支持迭代, 内存占用小


# 保存python对象到文件
X,X,Y=43,44,45
S='Spam'
L=[1,2,3]
D={'a':1,'b':2}

F=open('datafile.txt','w')
F.write(S+'\n')
F.write('%s,%s,%s\n' % (X,Y,Z))   数字对象转换成字符串
F.write(str(L) + '$' + str(D) + '\n')   转换列表和字典成字符串
F.close()


# 从文件读取python对象
F=open('datafile.txt')
读取包含字符串的行
line=F.readline()
line.rstrip()  去掉多余行终止符
读取包含数字的行
line=F.readline()
parts=line.split(',')
int(parts[0])
X,Y,Z=[int(P) for P in parts]
读取列表和字典
line=F.readline()
parts=line.split('$')
L=eval(parts[0])
D=eval(parts[1])


# pickle存储原生对象
# 存储:
D={'a':1,'b':2}
F=open('datafile.pkl','wb')
import pickle
pickle.dump(D,F)     dump(python对象,文件handle)
F.close
# 读取:
F=open('datafile.pkl','rb')
D=pickle.load(F)
```
