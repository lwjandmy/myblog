# 学习_ENV和_G

```text
-- 对一个全局块，load函数在加载此块时，执行了
-- local _ENV = _G
-- return function(...)

--    ...
-- end
-- 即load函数将_ENV放在闭包中,将变量限制在_ENV中

var = 1
fun = loadfile("1.lua", "bt", {print = print, _G = { var = 3 }, var = 2});
fun()

-- 1.lua:
-- print(var)
-- print(_G.var)
```
